# ============================================================
# PROCESO: Ciclo de inicio de sesión
# ============================================================

# Bandera que controla si ya se logró entrar al sistema
sesion_activa ← Falso

MIENTRAS sesion_activa sea igual a Falso HACER

    # Pedir las credenciales de acceso
    usuario_ingresado ← pedir al usuario "Usuario: "
    clave_ingresada ← pedir al usuario "Contraseña: "

    # Verificar que el usuario exista y que la contraseña coincida
    SI usuario_ingresado está registrado en usuarios Y la contraseña guardada para ese usuario es igual a clave_ingresada ENTONCES

        sesion_activa ← Verdadero

    FIN SI

FIN MIENTRAS

imprimir "Bienvenido, " + nombre_usuario + " (" + rol_usuario + ")"

# ============================================================
# DECISIÓN: Ramificación según el rol de quien inició sesión
# ============================================================

SI rol_usuario es igual a "alumno" ENTONCES

    imprimir "Boleta de " + nombre_usuario

    # Conjunto donde se guardarán las materias con calificación aprobatoria
    materias_aprobadas ← conjunto vacío

    PARA CADA materia_actual EN materias HACER

        calificacion_actual ← calificación de usuario_ingresado en materia_actual

        imprimir materia_actual + ": " + calificacion_actual

        SI calificacion_actual es mayor o igual a 8.0 ENTONCES
            agregar materia_actual a materias_aprobadas
        FIN SI

    FIN PARA

    # Las materias pendientes son todas las que no quedaron en aprobadas
    materias_pendientes ← conjunto(materias) menos materias_aprobadas

    SI materias_aprobadas no está vacío ENTONCES
        imprimir "Materias aprobadas: " + materias_aprobadas
    SINO
        imprimir "Materias aprobadas: Ninguna"
    FIN SI

        imprimir "Materias pendientes: " + materias_pendientes
    SINO
        imprimir "Materias pendientes: Ninguna"
    FIN SI

SINO SI rol_usuario es igual a "maestro" ENTONCES

    FIN PARA

    # Variable que controla si el maestro desea seguir calificando

    MIENTRAS seguir_calificando sea igual a 's' HACER

        alumno_seleccionado ← pedir al usuario "Alumno (usuario): "

        MIENTRAS alumno_seleccionado no esté en calificaciones HACER

            materia_seleccionada ← pedir al usuario "Materia: "
        FIN MIENTRAS

        calificacion_nueva ← pedir al usuario "Nueva calificación: " (convertir a número decimal)

    FIN MIENTRAS

SINO SI rol_usuario es igual a "coordinador" ENTONCES

    PARA CADA clave_usuario EN usuarios HACER

            imprimir clave_usuario + " - " + nombre guardado para clave_usuario
        FIN SI
    FIN PARA
